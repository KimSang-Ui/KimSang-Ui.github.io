# 1부터 10까지의 합을 for구문과 while 구문으로 표현하기

for구문

public class Sum 1 to 10 {
	
	public static void main(String[] args) {
		
  int sum = 0;
	
  for(int i = 1; i<=10; i++) {
	
   sum+=i;
			
			
		}
		System.out.println(sum);
}

}

while구문

public class Sum 1 to 10 {
	
	
 public static void main(String[] args) {
	
  int sum = 0;
		
  int i = 1;
	
  while(i<=10){
	
   sum+=i;
			
			
		}
		System.out.println(sum);
}

}
