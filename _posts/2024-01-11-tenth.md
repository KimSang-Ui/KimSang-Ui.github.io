# for문과 scanner를 이용한 구구단 
import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		System.out.println("단을 입력하세요");
		int a = scan.nextInt();	
		for(int j = 1; j <=9; j++) {
			System.out.println(a+"x"+j +"="+(a*j));
		}
		System.out.println();
		}	
		}
